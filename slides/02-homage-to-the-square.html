Remix time! Creating a small system

make_homage <- function(colors = c("#5A9CBA", "#919EA3", "#F1EFDF", "#F5BB1D")) {
  tribble(
   ~x0, ~y0, ~size,
     0,   0,    10,
     1, 0.5,     8,
     2,   1,     6,
     3, 1.5,     4
   ) %>%
    mutate(
      x1 = x0 + size,
      y1 = y0 + size,
      color = sample(colors, size = 4)
    ) %>%
    ggplot() +
    geom_rect(aes(
      xmin = x0, xmax = x1,
      ymin = y0, ymax = y1,
      fill = color
    )) +
    scale_fill_identity() +
    coord_fixed() +
    theme_void()
}

extract_hsl <- function(data) {
  data %>%
    mutate(
      color = color(color),
      h = clr_extract_hue(color),
      s = clr_extract_saturation(color),
      l = clr_extract_lightness(color)
    ) %>%
    select(color, h, s, l)
}

homage_hsl <- homage %>%
